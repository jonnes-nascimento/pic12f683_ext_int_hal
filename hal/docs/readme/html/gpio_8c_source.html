112/************************************************************************************************
113 * Function: void Gpio_Init(const GpioConfig_t * const Config);
114 */
154void Gpio_Init(const GpioConfig_t * const Config)
155{
156 /* This is for test only - this code must be moved */
157 *SystemOscilator = 0x70;
158 *SystemComparator = 0x07;
159 *SystemADConverter = 0x00;
161 // Disable all interrupts
162 *InterruptControl[GPIO_PORT] = 0x00;
163 *PeripheralInterruptEnable[GPIO_PORT] = 0x00;
164 *PeripheralInterruptRequest[GPIO_PORT] = 0x00;
165
166 for (uint8_t port = 0; port < GPIO_PORT_COUNT; port++)
167 {
168 *GpioPort[port] = 0x00;
169 *GpioFunction[port] = 0x00;
171 /* Sets the GPIOs */
172 for (uint8_t i = 0; i < GPIO_CHANNEL_COUNT; i++)
173 {
174 switch (Config[i].Function)
175 {
176 case GPIO_PIN_FUNCTION_DIGITAL:
177 // TRISIO: set de pin direction mode
178 if (Config[i].Direction == GPIO_PIN_DIRECTION_INPUT)
179 {
180 *GpioDirection[port] |= (Config[i].Direction << i);
181 }
182 else
183 {
184 *GpioDirection[port] &= ~(HIGH << i);
185 }
186 break;
187 case GPIO_PIN_FUNCTION_CAPTURE_COMPARE: break;
188 case GPIO_PIN_FUNCTION_COMPARATOR: break;
189 case GPIO_PIN_FUNCTION_INTERRUPT_ON_CHANGE: break;
190 case GPIO_PIN_FUNCTION_TIMER: break;
191 case GPIO_PIN_FUNCTION_ANALOG:
192 *GpioFunction[port] |= (HIGH << i);
193 break;
194 case GPIO_PIN_FUNCTION_EXTERNAL_INTERRUPT:
195 // ANSEL as digital I/O
196 *GpioFunction[port] &= ~(HIGH << i);
197
198 // TRISIO set de direction of pin as INPUT
199 *GpioDirection[port] |= (Config[i].Direction << i);
200
201 // Sets the edge trigger of the interrupt
202 if (Config[i].InterruptEdge == GPIO_INTERRUPT_EDGE_FALLING)
203 {
204 *GpioOption[port] &= ~INTEDG_BIT_MASK;
205 }
206 else if (Config[i].InterruptEdge == GPIO_INTERRUPT_EDGE_RISING)
207 {
208 *GpioOption[port] |= INTEDG_BIT_MASK;
209 }
210 else
211 {
212 // do nothing
213 }
214
215 // Clear the external interrupt flag INTF
216 *InterruptControl[port] &= ~INTF_BIT_MASK;
217
218 // Enable external interrupts
219 *InterruptControl[port] |= INTE_BIT_MASK;
220
221 // Enable global interrupts
222 *InterruptControl[port] |= GIE1_BIT_MASK;
223
224 // Save the callback function executor address to the FSR register
225 //*FSR = ?
226 //asm volatile ("JMP *%0" : : "r" (&Gpio_Interrupt_Callback_Executor));
227
228 break;
229 } // switch Config[i].Function
230
231 // sets the GPIO initial state
232 *GpioPort[port] |= (Config[i].State << i);
233
234 } // for GPIO_CHANNEL_COUNT
235 } // for GPIO_PORT_COUNT
236} // Gpio_Init
